1. Introduction:

        The main idea behind the final project of CECS 561 Hardware/Software co-design course was to build a web server on Zybo board which can host lightweight web application and the number of devices which are connected in the same network can view this web application on their device. Final Project is consist of two phases. The phases of the Final Project incorporated a different functional feature to implement the final phase of the Final Project which was an “online Chess Board Game”. Each phase of the Final Project will be explained along with a brief introduction of the Xilinx generated files that were used.

Phase 1 - Description:

           Phase one is the first part of final project. The initial target was to host website on a Zybo board to control LEDs and also to check the status of switches and buttons on different devices like computers or mobile phones. 2. Operation:

Fig. - Operation Flow

Phase 1 - Operation:

To set up the project following steps are required before working.

  1. Before powering up the board, first connect all the hardware components together by connecting ethernet cable to board and wifi router, connect power supply cable to board and wifi router, insert micro-SD card in board slot which contains bootable file and enable micro-SD pin onboard.
  2. Now power up the board and wifi router. On board ethernet light will start blinking and Petalinux will boot in 30 sec.
  3. By using any wifi support device like mobile phone or laptop connect to the wifi router and type board IP in browser URL “192.168.1.54:8080”.
  4. Zybo Controller page will be loaded on browser and from here LED can be turned ON/OFF and can see live updating status of Switches and Buttons on page.

Phase 2 - Operation:

  1. Connect Zybo board with UART cable.
  2. Connect device (mobile/laptop) with common wifi network and type                           http://192.168.1.54:8080 in browser and press Enter. (this will send the request to zybo board)
  3. Zybo board will accept the request and opens a new game on device’s screen.
  4. Now we can play online Chess game on your device, where the opponent is zybo board.

#include "DevUIOMyIP.h"

#define TIMEOUT (200)        // 200msec

DevUIOMyIP *pDevUIOMyIP;

// Read data from pipe buffer “pipe_p2c” which is written by web server on request of devices. This pipe buffer is used for led control.

int recvPipe(char *buf, int bufNumMax)

{

        int len = 0;

        int  fdP2C;        // pipe from Python to C  (read)

        if ((fdP2C = open("/home/root/pipe_p2c", O_RDWR))<0) {

                perror("open");

                exit(-1);

        }

        struct pollfd pfd;

        pfd.fd = fdP2C;

        pfd.events = POLLIN;

        if(poll(&pfd, 1, TIMEOUT) != 0) {

                len = read(fdP2C, buf, bufNumMax - 1);

                buf[len] = '\0';

        } else {

                printf("timeout\n");

        }

        close(fdP2C);

        return len;

}

// Write data in pipe buffer “pipe_c2p” which is read by web server and update the status of all the devices that are connected in wifi network. This is used for Switch and Button status.

void sendPipe(char *buf)

{

        int  fdC2P;        // pipe from C to Python  (write)

        if ((fdC2P = open("/home/root/pipe_c2p", O_RDWR))<0) {

                perror("open");

                exit(-1);

        }

        write(fdC2P, buf, strlen(buf));

        close(fdC2P);

}

// treatCmd() function write the LED registers values by calling setLedbit() which is define in pDevUIOMyIP.h. The value which is writen is the value that is  read by recvPipe(from devices).

void treatCmd(char *buf)

{

        /* command format = "LD0,LD1,LD2,LD3" */

        /* e.g. "ON,OFF,ON,OFF" */

        char separators[] = ",";

        char *prm;

        if ((prm = strtok(buf, separators)) == NULL) {

printf("err: %s\n", buf); return;

}

        for (int indexLed = 0; indexLed < 4; indexLed++) {

                if (strcmp(prm, "ON") == 0) {

                        pDevUIOMyIP->setLedBit(indexLed);

                } else if (strcmp(prm, "OFF") == 0) {

                        pDevUIOMyIP->clrLedBit(indexLed);

                } else {

                        printf("err: %s\n", buf);

                }

                if ((prm = strtok(NULL, separators)) == NULL) {printf("err: %s\n", buf); return;}

        }

}

File DevUIOMyIP.cpp:-

#include "DevUIOMyIP.h"

// Initialized initial led value to 0 : to turn OFF all LEDs.

DevUIOMyIP::DevUIOMyIP(const char* dev) : DevUIO(dev)

{

        writeReg(REG_OFFSET_LED, 0x00);

}

// Write LED register value.

void DevUIOMyIP::setLed(uint32_t led)

{

        writeReg(REG_OFFSET_LED, led & 0x0F);

}

// Write particuler LED register using its index value (Turn ON).

void DevUIOMyIP::setLedBit(uint32_t bit)

{

        uint32_t reg = readReg(REG_OFFSET_LED);

        writeReg(REG_OFFSET_LED, reg | (1 << bit));

}

// Write particuler LED register using its index value (Turn OFF).

void DevUIOMyIP::clrLedBit(uint32_t bit)

{

        uint32_t reg = readReg(REG_OFFSET_LED);

        writeReg(REG_OFFSET_LED, reg & ~(1 << bit));

}

// Read register value of the button.

uint32_t DevUIOMyIP::getBtn()

{

        return readReg(REG_OFFSET_BTN) & 0x0F;

}

//Read register value of Switch.

uint32_t DevUIOMyIP::getSw()

{

        return readReg(REG_OFFSET_SW) & 0x0F;

}
